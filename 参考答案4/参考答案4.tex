\subsection{16.1.2}

\begin{enumerate}[(1)]
	\item $u=\sqrt[z]{\frac{x}{y}}$ at $(1,1,1)$.
	\par $\pd{u}{x}(1,1,1)=\frac{\dif x}{\dif x}\big|_{x = 1} = 1. $
	\par $\pd{u}{y}(1,1,1)=\left(\frac{1}{ y}\right)'\big|_{y = 1} = -1. $
	\par $\pd{u}{z}(1,1,1)=\frac{\dif 1}{\dif z}\big|_{z = 1} = 0. $
	\item $z=x+(y-1)\arcsin \frac{x}{y}$ at $(0,1)$.
	\par $\pd{z}{x}(0,1)=\frac{\dif x}{\dif x}\big|_{x = 0} = 1. $
	\par $\pd{z}{y}(0,1)=\frac{\dif 0}{\dif y}\big|_{y = 1} = 0. $
	\item $u=\arctan \frac{x+y+z-xyz}{1-xy-xz-yz}$ at $(0,0,0)$.
	\par $\pd{u}{x}(0,0,0)=\left(\arctan x\right)'\big|_{x = 0} = 1. $
	\par $\pd{u}{y}(0,0,0)=\left(\arctan y\right)'\big|_{y = 0} = 1. $
	\par $\pd{u}{z}(0,0,0)=\left(\arctan z\right)'\big|_{z = 0} = 1 . $

\end{enumerate}

\subsection{16.2.4}

$u(x+h,y+k)-u(x,y)=Ah+Bk+o(\sqrt{h^2+k^2})$,
\par $v(x+h,y+k)-v(x,y)=Ch+Dk+o(\sqrt{h^2+k^2})$.
\par $\Delta uv = u(x+h,y+k)v(x+h,y+k) - u(x,y)v(x,y)$
\par $\phantom{\Delta uv}=[ u(x+h,y+k)v(x+h,y+k) - u(x,y) v(x+h,y+k) ]+[ u(x,y)v(x+h,y+k) - \phantom{\Delta uv=\ }u(x,y)v(x,y) ]$
\par $\phantom{\Delta uv}=[ (Ah+Bk)v(x+h,y+k) + v(x+h,y+k) o(\sqrt{h^2+k^2})  ]+[ u(x,y)(Ch+Dk) + \phantom{\Delta uv=\ }u(x,y) o(\sqrt{h^2+k^2}) ]$,
\par $\because \lim \limits_{\sqrt{h^2+k^2} \to 0} v(x+h,y+k)= v(x,y),$
\par $\therefore \dif \, (uv) = v\dif u + u \dif v.$\qed

\end{enumerate}
