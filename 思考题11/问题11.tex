\subsection{1}

设一元函数 $f(x)$ 在 $[a, b]$ 上可积.在 $[a, b] \times[a, b]$ 上定义 $F(x, y)=[f(x)-f(y)]^{2}$
\begin{enumerate}[(1)]
	\item 将重积分 $\iint_{D} F(x, y) \dif x \dif y$ 化为累次积分;
	\item 证明: $\left[\int_{a}^{b} f(x) \dif x\right]^{2} \leqslant (b-a) \int_{a}^{b} f^{2}(x) \dif x$ .
\end{enumerate}
